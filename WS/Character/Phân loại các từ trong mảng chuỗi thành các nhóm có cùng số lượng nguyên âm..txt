    private static final Set<Character> VOWELS = Set.of('a', 'e', 'i', 'o', 'u');
    private static int countDistinctVowels(String word) {
        Set<Character> vowelsFound = new HashSet<>();
        for (char c : word.toLowerCase().toCharArray()) {
            if (VOWELS.contains(c)) {
                vowelsFound.add(c);
            }
        }
        return vowelsFound.size();
    }
    public static List<String> groupWordsByDistinctVowelCount(List<String> words) {
        Map<Integer, List<String>> groupedMap = words.stream().collect(Collectors.groupingBy(
                        word -> countDistinctVowels(word), // Hàm dùng để phân loại
                        TreeMap::new,                   // Sử dụng TreeMap để sắp xếp nhóm
                        Collectors.toList()             // Tập hợp các từ vào một List
                ));
        return groupedMap.values().stream().map(groupList -> {
                    Collections.sort(groupList);                    
                    return String.join(", ", groupList);
                }).collect(Collectors.toList());
    }
