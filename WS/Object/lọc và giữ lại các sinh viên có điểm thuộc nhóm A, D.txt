import java.util.stream.Collectors;
    public static List<Student> locSinhVienDiemAD(List<Student> studentList) {
        // Xử lý trường hợp danh sách đầu vào là null
        if (studentList == null) {
            return new ArrayList<>(); // Trả về danh sách rỗng
        }

        // Sử dụng Stream API
        List<Student> ketQua = studentList.stream()
            .filter(student -> student.getScore() >= 8.0f || student.getScore() < 5.0f)
            .collect(Collectors.toList());

        return ketQua;
    }
