    public static byte[] maHoaCaesarBytes(byte[] data) {
        int shift = data.length % 256;
        byte[] ketQua = new byte[data.length];
        for (int i = 0; i < data.length; i++) {
            byte b = data[i];
            int giaTriKhongDau = b & 0xFF;
            int giaTriMoi = (giaTriKhongDau + shift) % 256;
            ketQua[i] = (byte) giaTriMoi;
        }
        return ketQua;
    }
